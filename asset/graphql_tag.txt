query ($owner: String!, $name: String!) {
  repository(owner: $owner, name: $name) {
    refs(first: 10, orderBy: {field: TAG_COMMIT_DATE, direction: DESC}, refPrefix: "refs/tags/") {
      nodes {
        name
        repository {
          url
        }
      }
    }
  }
  rateLimit(dryRun: false) {
    cost
    limit
    nodeCount
    remaining
    resetAt
  }
}
