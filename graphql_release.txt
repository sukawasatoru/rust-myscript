query ($owner: String!, $name: String!) {
  repository(owner: $owner, name: $name) {
    releases(first: 10, orderBy: {field: CREATED_AT, direction: DESC}) {
      nodes {
        name
        tag {
          name
        }
        isDraft
        isPrerelease
        publishedAt
        url
      }
    }
  }
  viewer {
    login
  }
  rateLimit(dryRun: false) {
    cost
    limit
    nodeCount
    remaining
    resetAt
  }
}
